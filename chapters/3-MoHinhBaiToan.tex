\section{Bài toán tự phục hồi trong Kubernetes}

\noindent Trong môi trường microservices, Kubernetes cung cấp cơ chế \textit{tự phục hồi} để tự động phục hồi dịch vụ khi xảy ra sự cố Pod hoặc Node. Tuy nhiên, cơ chế mặc định chủ yếu dựa trên các quy tắc cố định và các controller truyền thống, do đó thường chưa tối ưu trong các tình huống tải động và lỗi kéo dài. Bài toán đặt ra là lựa chọn hành động phục hồi phù hợp theo từng trạng thái hệ thống, nhằm giảm lỗi dịch vụ, giảm độ trễ và rút ngắn thời gian phục hồi.

\noindent Trong phạm vi đề tài, bài toán được mô hình hóa dưới dạng tiến trình quyết định Markov (Markov Decision Process - MDP), trong đó tác nhân học tăng cường quan sát trạng thái hệ thống và quyết định hành động phục hồi theo thời gian.

\section{Thiết kế MDP cho các kịch bản lỗi}

\noindent Một MDP được biểu diễn bởi bộ $(\mathcal{S}, \mathcal{A}, \mathcal{P}, \mathcal{R}, \gamma)$, trong đó:
\begin{itemize}
    \item $\mathcal{S}$: không gian trạng thái hệ thống.
    \item $\mathcal{A}$: tập hành động phục hồi.
    \item $\mathcal{P}$: hàm chuyển trạng thái.
    \item $\mathcal{R}$: hàm thưởng.
    \item $\gamma$: hệ số chiết khấu.
\end{itemize}

\noindent Phạm vi hiện tại gồm hai kịch bản lỗi độc lập, do đó sử dụng hai MDP tách biệt:
\begin{itemize}
    \item \textbf{MDP\_PPO\_POD\_KILL}: cho kịch bản \texttt{chaos-pod-kill-product}.
    \item \textbf{MDP\_PPO\_NETWORK\_DELAY}: cho kịch bản \texttt{chaos-network-delay-api-gateway}.
\end{itemize}

\noindent\textbf{Hành động.}

\noindent Trong scope hiện tại, tập hành động an toàn gồm 4 hành động rời rạc:
\begin{itemize}
    \item $a_0$: \texttt{idle}
    \item $a_1$: \texttt{restart\_pod}
    \item $a_2$: \texttt{scale\_up}
    \item $a_3$: \texttt{scale\_down}
\end{itemize}


\noindent\textbf{Hàm thưởng.}

\noindent Hàm thưởng cho \texttt{MDP\_PPO\_POD\_KILL} được thiết kế như sau:
\[
R_t = 6.0 \cdot health\_gain - 3.0 \cdot err\_{5xx} - 2.0 \cdot p99\_norm - 1.2 \cdot tail\_stress - action\_cost - 0.05
\]
\noindent trong đó:
\[
health\_gain = health_t - health_{t-1}
\]
\[
health = 0.45(1-err\_{5xx}) + 0.35(1-p99\_norm) + 0.20 \cdot http\_rps\_norm
\]

\noindent Chi phí hành động:
\begin{itemize}
    \item \texttt{idle}: 0.00
    \item \texttt{restart\_pod}: 0.30
    \item \texttt{scale\_up}: 0.15
    \item \texttt{scale\_down}: 0.20
\end{itemize}

\noindent Thưởng/phạt bổ sung:
\begin{itemize}
    \item \textbf{stability bonus}: $+8$ nếu hệ ổn định liên tục $k$ bước $(err\_{5xx}<0.02 \land p99\_norm<0.25)$
    \item \textbf{SLA violation penalty}: $-10$ nếu $(err\_{5xx}>0.1 \lor p99\_norm>0.9)$ kéo dài $\ge 3$ bước
\end{itemize}

\noindent\textbf{Điều kiện kết thúc một tập (episode).}

\noindent Tập huấn luyện kết thúc khi một trong ba điều kiện xảy ra:
\begin{itemize}
    \item \textbf{Thành công}: hệ ổn định liên tục $k$ bước.
    \item \textbf{Thất bại}: trạng thái vi phạm SLA kéo dài theo ngưỡng định nghĩa.
    \item \textbf{Hết thời gian}: đạt $max\_steps$.
\end{itemize}

\noindent\textbf{Mô hình chuyển trạng thái.}

\noindent Động lực học của môi trường gồm:
\begin{itemize}
    \item \textbf{Ngoại sinh}: lấy từ chuỗi thời gian của tệp kết quả \texttt{combined-report.csv}.
    \item \textbf{Nội sinh}: do hành động của tác nhân tác động lên $err/p99/rps$, sau đó blend về quỹ đạo artifact để tránh overfit.
\end{itemize}

\subsection{Kịch bản độ trễ mạng tăng cao}

\noindent Kịch bản độ trễ mạng tăng cao được mô hình hóa bằng \textbf{MDP\_PPO\_NETWORK\_DELAY}. Cấu trúc MDP giữ nguyên, nhưng điều chỉnh ưu tiên:
\begin{itemize}
    \item tăng trọng số phạt cho $p95/p99$ latency,
    \item giảm tác dụng kỳ vọng của \texttt{restart\_pod},
    \item tăng tác dụng kỳ vọng của \texttt{scale\_up} khi suy giảm mạng gây nghẽn.
\end{itemize}
